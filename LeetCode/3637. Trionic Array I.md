## 題解  

```cpp=
class Solution {
public:
    bool isTrionic(vector<int>& nums) {
        int n = nums.size();
        if(n < 4)   return false;
        int reverse = 0;
        int pre = 0;
        for(int i = 1; i < n; ++i) {
            if(nums[i] == nums[i-1])    return false;
            if(reverse % 2 == 0) {
                // increase
                if(nums[i] < nums[i-1]) {
                    if(i - pre == 1)    return false;
                    pre = i-1;
                    reverse += 1;
                }
            } else {
                // decrease
                if(nums[i] > nums[i-1]) {
                    if(i - pre == 1)    return false;
                    pre = i-1;
                    reverse += 1;
                }
            }
        }
        return (reverse == 2);
    }
};
```

Time Complexity : $O(n)$  
Space Complexity : $O(1)$
