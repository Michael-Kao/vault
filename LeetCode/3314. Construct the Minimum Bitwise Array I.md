## 題解  

```cpp=
class Solution {
public:
    vector<int> minBitwiseArray(vector<int>& nums) {
        int n = nums.size();
        vector<int> res(n, -1);
        for(int i = 0; i < n; ++i) {
            int target = nums[i];
            for(int j = 0; j < target; ++j) {
                if((j | j+1) == target) {
                    res[i] = j;
                    break;
                }
            }
        }
        return res;
    }
};
```

Time Complexity : $O(mn)$  
Space Complexity : $O(n)$
