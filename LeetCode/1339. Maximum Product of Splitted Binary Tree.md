# [1339. Maximum Product of Splitted Binary Tree](https://leetcode.com/problems/maximum-product-of-splitted-binary-tree/description/)  

+ 日期 : 2026/01/07  

## 問題描述  

基礎 `tree traversal`  

## 直覺想法  

1. `DFS` 找出子樹的總和  
2. 然後找出兩個子樹的乘積最大值  

## 題解  

```cpp=
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxProduct(TreeNode* root) {
        int MOD = 1e9 + 7;
        vector<int> arr;
        int total = dfs(root, arr);
        long long cur = 0;
        for(int num : arr) {
            if((1ll * (total - num) * num) > cur) {
                cur = ((1ll * (total - num) * num));
            }
        }
        return cur % MOD;
    }

    int dfs(TreeNode *node, vector<int> &arr) {
        if(!node)   return 0;
        int left = dfs(node->left, arr);
        int right = dfs(node->right, arr);

        arr.push_back(left + node->val + right);
        return left + node->val + right;
    }
};
```

Time Complexity : $O(n)$  
Space Complexity : $O(n)$
