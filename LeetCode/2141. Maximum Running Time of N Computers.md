## 題解  

```cpp=
class Solution {
public:
    long long maxRunTime(int n, vector<int>& batteries) {
        sort(batteries.begin(), batteries.end());
        long long extra = 0;
        for(int i = 0; i < batteries.size()-n; ++i) {
            extra += batteries[i];
        }

        for(int i = batteries.size()-n; i < batteries.size()-1; ++i) {
            if(extra / (i+1-batteries.size()+n) < batteries[i+1] - batteries[i])
                return (long long)batteries[i] + extra / (i+1-batteries.size()+n);
            extra -= (long long)(i+1-batteries.size()+n) * (batteries[i+1] - batteries[i]);
        }
        return (long long)batteries.back() + extra / n;
    }
};
```

Time Complexity : $O(mlogm)$  
Space Complexity : $O(m)$
