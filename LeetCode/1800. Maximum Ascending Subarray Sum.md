```cpp=
class Solution {
public:
    int maxAscendingSum(vector<int>& nums) {
        int res = 0;
        int cur = nums[0];
        for(int i = 1; i < nums.size(); ++i) {
            if(nums[i] <= nums[i - 1]) {
                res = max(res, cur);
                cur = 0;
            } 
            cur += nums[i];
        }
        res = max(res, cur);
        return res;
    }
};
```

Time Complexity : $O(n)$  
Space Complexity : $O(1)$
