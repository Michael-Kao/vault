```cpp=
class Solution {
public:
    int minDeletionSize(vector<string>& strs) {
        int m = strs[0].size();
        int n = strs.size();
        vector<int> dp(m, 1);
        for(int i = m-2; i >= 0; --i) {
            for(int j = i+1; j < m; ++j) {
                bool check = true;
                for(int k = 0; k < n; ++k) {
                    if(strs[k][i] > strs[k][j]) {
                        check = false;
                        break;
                    }
                }
                if(check) {
                    dp[i] = max(dp[i], 1 + dp[j]);
                }
            }
        }
        int maxC = 0;
        for(int i = 0; i < m; ++i)
            maxC = max(maxC, dp[i]);
        return m - maxC;
    }
};
```

Time Complexity : $O(nm^2)$  
Space Complexity : $O(m)$
